Algorithm SentenceAnalysis
Variables
    ch: character
    length, words, vowels: integer

Begin
    length ← 0
    words ← 1
    vowels ← 0

    Read ch
    While ch ≠ '.'
        length ← length + 1

        If ch = ' ' Then
            words ← words + 1
        EndIf

        If ch = 'a' OR ch = 'e' OR ch = 'i' OR ch = 'o' OR ch = 'u' 
           OR ch = 'A' OR ch = 'E' OR ch = 'I' OR ch = 'O' OR ch = 'U' Then
            vowels ← vowels + 1
        EndIf

        Read next ch
    EndWhile

    length ← length + 1   // to count the period

    Write "Length: ", length
    Write "Words: ", words
    Write "Vowels: ", vowels
End
